### demo {#demo}

简单看一个 carla 编写应用的 demo，是不是发现有些眼熟。

```javascript
import { Home } from "./page/Home";
import { DemoInfo } from "./page/DemoInfo";
import { DemoContent } from "./page/DemoContent";
import { search as searchApi } from "./search";
import { Page, CarlaUI } from "carla";

export const uiEntry = function (context) {
  return CarlaUI({
    namespace: "century-comic",
    context,
    pages: {
      home: Page({
        entry: true,
        page: Home,
        title: "漫画",
      }),
      info: Page({
        page: DemoInfo,
        title: "路由",
      }),
      content: Page({
        page: DemoContent,
        title: "demo",
      }),
    },
  });
};
```
